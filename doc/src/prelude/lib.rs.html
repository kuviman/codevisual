/// Sort two values.
///
/// # Examples
/// ```
/// use prelude::*;
/// assert_eq!(min_max(6, 3), (3, 6));
/// ```
pub fn min_max<T: PartialOrd>(a: T, b: T) -> (T, T) {
    if a < b {
        (a, b)
    } else {
        (b, a)
    }
}

/// Get minimum of two values.
///
/// # Examples
/// ```
/// use prelude::*;
/// assert_eq!(min(1, 2), 1);
/// ```
pub fn min<T: PartialOrd>(a: T, b: T) -> T {
    min_max(a, b).0
}

/// Get maximum of two values.
///
/// # Examples
/// ```
/// use prelude::*;
/// assert_eq!(max(1, 2), 2);
/// ```
pub fn max<T: PartialOrd>(a: T, b: T) -> T {
    min_max(a, b).1
}
